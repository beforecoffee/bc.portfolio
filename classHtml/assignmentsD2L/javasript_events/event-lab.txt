	var bodyElm = document.body;
	bodyElm = addEventListener("load",loadFunction);

function loadFunction() {

	//some mouse-related events
	var myImg = document.getElementById("guitarImg");
	myImg.addEventListener("mouseover",mouseStuff);
	myImg.addEventListener("mouseout",mouseStuff);
	myImg.addEventListener("mouseup",mouseStuff);
	myImg.addEventListener("mousedown",mouseStuff);
	
	// click-related
	var clickSection = document.getElementById("clicky");
	clickSection.addEventListener("click",clickHandler);
	clickSection.addEventListener("dblclick",clickHandler);
	
	var textIn = document.getElementById("textInput");
	textIn.addEventListener("focus",inputHandler);
	textIn.addEventListener("blur",inputHandler);
	
	var selectIn = document.getElementById("selectInput");
	selectIn.addEventListener("change",selectHandler);

}  // end function loadFunction


function mouseStuff(evt) {
	//console.log(evt);
	
	switch(evt.type) {
		case "mouseover":
			console.log('you moused over');
			break;
		case "mouseout":
			console.log('bye bye mouse');
			break;
		case "mousedown":
			console.log('ouch');
			break;
		case "mouseup":
			console.log('you let up');
			break;
	}
}  // end function mouseStuff



function inputHandler(evt) {
	if (evt.type == "focus") {
		console.log("you are typing into the form");
	} else if (evt.type == "blur") {
		console.log("you are done typing");
		//you could do early validation here, or a lookup:
		console.log(evt.target.value);
	}	
} //end function inputHandler



function selectHandler(evt) {
	var myDiv = document.getElementById("selectResult");
	myDiv.innerHTML = evt.target.value;
} //end function selectHandler



function clickHandler(evt) {

	if (evt.type == "click") {
		console.log("you clicked once");
	} else if (evt.type == "dblclick") {
		console.log("you dbl clicked");
	}
	
} //end function clickHandler
